59 // this is the ATCACommand parameter structure. The caller to the command method must
60 // initialize param1, param2 and data if appropriate. The command method will fill in the rest
61 // and initialize the packet so it's ready to send via the ATCAIFace.
62 // this particular structure mimics the ATSHA and ATECC family device's command structures
63 
64 // Note: pack @ 2 is required, @ 1 causes word alignment crash (though it should not), a known bug in GCC.
65 // @2, the wire still has the intended byte alignment with arm-eabi. this is likely the least portable part of atca
66 
67 
68 #ifdef ATCA_NO_PRAGMA_PACK
69 typedef struct __attribute__ ((packed))
70 #else
71 #pragma pack( push, ATCAPacket, 2 )
72 typedef struct
73 #endif
77 {
78 
79  // used for transmit/send
80  uint8_t _reserved; // used by HAL layer as needed (I/O tokens, Word address values)
81 
82  //--- start of packet i/o frame----
83  uint8_t txsize;
84  uint8_t opcode;
85  uint8_t param1; // often same as mode
86  uint16_t param2;
87  uint8_t data[192]; // includes 2-byte CRC. data size is determined by largest possible data section of any
88  // command + crc (see: x08 verify data1 + data2 + data3 + data4)
89  // this is an explicit design trade-off (space) resulting in simplicity in use
90  // and implementation
91  //--- end of packet i/o frame
92 
93  // used for receive
94  uint8_t execTime; // execution time of command by opcode
95 
96  // structure should be packed since it will be transmitted over the wire
97  // this method varies by compiler. As new compilers are supported, add their structure packing method here
98 
99 } ATCAPacket;
